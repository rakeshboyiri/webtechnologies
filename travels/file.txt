database name travel_db


CREATE TABLE buses (
id INT AUTO_INCREMENT PRIMARY KEY,
source VARCHAR(50),
destination VARCHAR(50),
vacant_seats INT
);
CREATE TABLE bookings (
id INT AUTO_INCREMENT PRIMARY KEY,
bus_id INT,
num_seats INT,
FOREIGN KEY (bus_id) REFERENCES buses(id)
);
